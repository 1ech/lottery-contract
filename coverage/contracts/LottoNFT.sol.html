        uint256[] memory amounts = new uint256[](_numberOfTickets);

        uint256[] memory tokenIDs = new uint256[](_numberOfTickets);
        for (uint32 i = 0; i < _numberOfTickets; i += 1) {

            tokenIDs[i] = tokenIDsCount_;
            amounts[i] = 1;
